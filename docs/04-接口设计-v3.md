## 2. createRecord（修改）

### 入参
```js
{
  imageUrl: string,
  mealType: 'breakfast' | 'lunch' | 'dinner' | 'snack',
  mainCategory: 'meal' | 'fast',    // 【v2 新增】正餐/速食
  foodCats: FoodCategory[],         // 【v3 新增】5 大食物类别，多选 ≥1
  note: string                      // ≤ 50 字

```

**FoodCategory 枚举**（v3 新增）：
```js
type FoodCategory = 'staple' | 'veggie' | 'protein' | 'soup' | 'dessert';
// staple   = 主食（米饭/面食/面包等）
// veggie   = 蔬菜水果
// protein  = 肉蛋水产
// soup     = 汤粥饮品
// dessert  = 甜品零食
```

### 出参
```js
{
  success: true,
  data: {
    recordId: 'xxx',
    status: 'on_time' | 'late',      // 【新增】服务器端计算
    createdAt: 1700000000000
  }
}
```

### 关键新增逻辑

```js
// 服务器端根据时间戳和餐次计算状态
function calcStatus(mealType, timestamp) {
  if (mealType === 'snack') return 'on_time';

  const d = new Date(timestamp);
  const hour = d.getHours();
  const min = d.getMinutes();
  const now = hour * 60 + min;  // 转为分钟

  const rules = {
    breakfast: { deadline: 10 * 60, lateDeadline: 11 * 60 },     // 600, 660
    lunch:     { deadline: 14 * 60, lateDeadline: 15 * 60 },     // 840, 900
    dinner:    { deadline: 21 * 60, lateDeadline: 22 * 60 }      // 1260, 1320
  };
  const r = rules[mealType];
  if (now <= r.deadline) return 'on_time';
  if (now <= r.lateDeadline) return 'late';
  return 'missed';   // 超过延期截止
}
```

**注意**：
- 加餐 (`snack`) 始终为 `on_time`
- 超过延期截止的 (`missed`) 仍允许保存，但显示"已缺卡"提醒（本期只展示，不阻断）
- v3 校验：`foodCats` 必须为长度 1~5 的非空数组，元素只能是 5 个枚举值之一

### 逻辑

```js
// 工具函数：判断日期是否为工作日
function isWorkDay(dateStr) {
  const d = new Date(dateStr);
  const day = d.getDay();
  return day >= 1 && day <= 5;  // 周一到周五
}

// 取本周工作日范围
function getWorkWeekRange(today = new Date()) {
  const dow = today.getDay();
  const monday = new Date(today);
  monday.setDate(today.getDate() - (dow === 0 ? 6 : dow - 1));
  monday.setHours(0, 0, 0, 0);
  const friday = new Date(monday);
  friday.setDate(monday.getDate() + 4);
  return {
    start: formatDate(monday),
    end: formatDate(friday),
    daysPassed: Math.min(
      Math.floor((today - monday) / 86400000) + 1,
      5
    )
  };
}
```

### 关键计算逻辑

```js
async function getWeeklySummary(openid, weekStart) {
  const range = getWorkWeekRange();
  const { start, end } = weekStart ? { start: weekStart, end: ... } : range;
  
  // 1. 拉取工作日内所有打卡
  const meals = await db.collection('meals').where({
    _openid: openid,
    date: db.command.gte(start).and(db.command.lte(end))
  }).get();
  
  // 2. 拉取周末打卡（仅展示用）
  // 周末 = 紧跟 workWeekEnd 的周六周日
  const weekendMeals = await db.collection('meals').where({
    _openid: openid,
    date: db.command.gte(weekEndSat).and(db.command.lte(weekEndSun))
  }).get();
  
  // 3. 计算
  const workDayMeals = meals.data;
  const totalMeals = workDayMeals.length;
  const mealCount = workDayMeals.filter(m => m.mainCategory === 'meal').length;
  const fastCount = workDayMeals.filter(m => m.mainCategory === 'fast').length;
  const onTimeCount = workDayMeals.filter(m => m.status === 'on_time').length;
  const lateCount = workDayMeals.filter(m => m.status === 'late').length;
  
  // 4. 缺卡：5 天 × 3 餐 - 实际打卡
  const expectedMeals = 5 * 3;
  const missedCount = Math.max(0, expectedMeals - totalMeals);
  
  // 5. Top3 标签
  const tagCounter = {};
  workDayMeals.forEach(m => {
    (m.tags || []).forEach(t => {
      tagCounter[t] = (tagCounter[t] || 0) + 1;
    });
  });
  const topTags = Object.entries(tagCounter)
    .sort((a, b) => b[1] - a[1])
    .slice(0, 3)
    .map(([name, count]) => ({
      name,
      count,
      rate: Math.round((count / totalMeals) * 100),
      emoji: TAG_EMOJI_MAP[name] || '🍽'
    }));
  
  // 6. 5 日统计
  const dailyStats = [];
  for (let i = 0; i < 5; i++) {
    const d = new Date(start);
    d.setDate(d.getDate() + i);
    const dateStr = formatDate(d);
    const dayMeals = workDayMeals.filter(m => m.date === dateStr);
    const isToday = dateStr === formatDate();
    const isFuture = d > new Date();
    
    if (isFuture) {
      dailyStats.push({
        date: dateStr, weekday: WEEKDAYS[i],
        onTime: 0, late: 0, missed: 0, total: 0,
        future: true
      });
    } else {
      const onTime = dayMeals.filter(m => m.status === 'on_time').length;
      const late = dayMeals.filter(m => m.status === 'late').length;
      const total = dayMeals.filter(m => m.mealType !== 'snack').length;  // 加餐不计
      const missed = Math.max(0, 3 - total);
      dailyStats.push({
        date: dateStr, weekday: WEEKDAYS[i],
        onTime, late, missed, total: total + dayMeals.filter(m => m.mealType === 'snack').length,
        isToday
      });
    }
  }
  
  return { ... };
}
```

## 6. sendLike（修改 · 每天 1 次）

### 改前
- 写 likes 集合，date 索引

### 改后
- 不写新集合
- 直接更新 users.likedPartnerDates 数组
- 检查今天是否已在数组中

```js
// 简化版
async function sendLike(openid) {
  const today = formatDate();
  const user = await getUser(openid);
  if (user.likedPartnerDates?.includes(today)) {
    return { success: false, code: 'ALREADY_LIKED' };
  }
  await db.collection('users').doc(user._id).update({
    data: {
      likedPartnerDates: db.command.push([today])
    }
  });
  return { success: true };
}
```
